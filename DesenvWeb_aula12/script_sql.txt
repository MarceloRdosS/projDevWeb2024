CREATE DATABASE IF NOT EXISTS bdveiculos ;
USE bdveiculos ;
CREATE TABLE IF NOT EXISTS `bdveiculos`.`tbveiculo` ( 
`placa` CHAR(7) NOT NULL , 
`fabricante` VARCHAR(50) NOT NULL , 
`marca` VARCHAR(50) NOT NULL , 
`ano` INT NOT NULL , 
`cor` VARCHAR(50) NOT NULL , 
PRIMARY KEY (`placa`)) ENGINE = InnoDB;
CREATE TABLE IF NOT EXISTS `bdveiculos`.`tbcliente` (
`codCliente` INT NOT NULL AUTO_INCREMENT ,
`nome` VARCHAR(50) NOT NULL , 
`idade` INT NOT NULL , 
`telefone` VARCHAR(10) NOT NULL , 
`carroPlaca` CHAR(7) NOT NULL ,
CONSTRAINT `fk_placa` FOREIGN KEY (`carroPlaca`) REFERENCES `tbveiculo`(`placa`) 
ON DELETE CASCADE ON UPDATE CASCADE,  
PRIMARY KEY (`codCliente`)) ENGINE = InnoDB;
CREATE TABLE IF NOT EXISTS `bdveiculos`.`tblogin` (
 `id` INT NOT NULL AUTO_INCREMENT ,
 `login` VARCHAR(30) NOT NULL ,
 `senha` VARCHAR(30) NOT NULL ,
 `nivel` INT NOT NULL ,
 PRIMARY KEY (`id`)) ENGINE = InnoDB;
